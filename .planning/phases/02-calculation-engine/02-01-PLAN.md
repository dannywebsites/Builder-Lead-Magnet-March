    ```typescript
    import { BUSINESS_RULES } from "./constants";
    import { roundCurrency } from "./currency";
    import type { CalculatorInput, CalculatorOutput, EntityType } from "./types";
    ```

    **calculateTaxBuffer(entityType: EntityType, grossPersonalDraw: number): number**
    - If entityType === "limited_company": return grossPersonalDraw / BUSINESS_RULES.CORP_TAX_BUFFER
    - Else: return grossPersonalDraw

    **calculateStaffCost(staffCount: number, staffHourlyRate: number, staffHoursPerWeek: number): { totalMonthlyHours: number; basePayroll: number; adjustedPayroll: number }**
    - totalMonthlyHours = staffCount * staffHoursPerWeek * BUSINESS_RULES.WEEKS_PER_MONTH
    - basePayroll = totalMonthlyHours * staffHourlyRate
    - adjustedPayroll = basePayroll * BUSINESS_RULES.EMPLOYER_BURDEN
    - Return all three

    **calculateBillableHours(totalMonthlyHours: number): number**
    - return totalMonthlyHours * BUSINESS_RULES.EFFICIENCY_CAP

    **calculateSlippage(directCostPct: number): number**
    - return directCostPct * BUSINESS_RULES.SLIPPAGE_FACTOR

    **calculateMRT(targetBusinessProfit: number, adjustedOverheads: number, realDirectCost: number): number**
    - return (targetBusinessProfit + adjustedOverheads) / (1 - realDirectCost)

    **calculate(input: CalculatorInput): CalculatorOutput**
    Compose all steps:
    1. targetBusinessProfit = calculateTaxBuffer(input.entityType, input.grossPersonalDraw)
    2. { totalMonthlyHours, adjustedPayroll } = calculateStaffCost(input.staffCount, input.staffHourlyRate, input.staffHoursPerWeek)
    3. totalBillableHours = calculateBillableHours(totalMonthlyHours)
    4. realDirectCost = calculateSlippage(input.directCostPct)
    5. adjustedOverheads = input.fixedOverheads + adjustedPayroll
    6. monthlyRevenueTarget = calculateMRT(targetBusinessProfit, adjustedOverheads, realDirectCost)
    7. monthlyBillings = monthlyRevenueTarget * (1 + input.vatRate)
    8. hourlyFloorRate = totalBillableHours > 0 ? monthlyRevenueTarget / totalBillableHours : 0
    9. jobsToWin = monthlyRevenueTarget / input.avgJobValue
    10. quotesNeeded = jobsToWin / BUSINESS_RULES.WIN_RATE
    11. leadsNeeded = quotesNeeded / BUSINESS_RULES.LEAD_CONVERSION_RATE

    Return CalculatorOutput with:
    - monthlyRevenueTarget: roundCurrency(monthlyRevenueTarget)
    - monthlyBillings: roundCurrency(monthlyBillings)
    - hourlyFloorRate: roundCurrency(hourlyFloorRate)
    - jobsToWin: Math.ceil(jobsToWin)
    - quotesNeeded: Math.ceil(quotesNeeded)
    - leadsNeeded: Math.ceil(leadsNeeded)
    - targetBusinessProfit: roundCurrency(targetBusinessProfit)
    - adjustedPayroll: roundCurrency(adjustedPayroll)
    - totalBillableHours: roundCurrency(totalBillableHours)
    - realDirectCost: realDirectCost (NOT rounded — it's a ratio, not currency)
    - adjustedOverheads: roundCurrency(adjustedOverheads)

    CRITICAL:
    - No `async` keyword anywhere

    - No class instances, no mutable state
    - Do NOT parameterize EFFICIENCY_CAP or SLIPPAGE_FACTOR — always read from BUSINESS_RULES
    - Do NOT round intermediate values — only round in the final return object
